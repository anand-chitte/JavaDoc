import java.util.*;
import java.util.stream.Collectors;

public class EmployeeManagementSystem {
    private final List<Employee> employeeList;

    public EmployeeManagementSystem() {
        employeeList = new ArrayList<>();
        employeeList.add(new Employee(1, "John Doe", "Male", 24, "Engineering", 2016, 75000));
        employeeList.add(new Employee(2, "Jane Smith", "Female", 28, "Marketing", 2018, 65000));
        employeeList.add(new Employee(3, "Alice Johnson", "Female", 29, "Engineering", 2015, 80000));
        employeeList.add(new Employee(4, "Bob Brown", "Male", 35, "Sales", 2019, 55000));
    }

    public static void main(String[] args) {
        EmployeeManagementSystem ems = new EmployeeManagementSystem();
        ems.queryEmployeeData();

       /* Stream.iterate(new long[]{0, 1}, f -> new long[]{f[1], f[0] + f[1]})
                .limit(10)
                .map(f -> f[0])
                .forEach(System.out::println);*/
    }

    private void queryEmployeeData() {
        Map<String, Long> genderCount = employeeList
                .stream()
                .collect(Collectors.groupingBy(employee -> employee.getGender(), Collectors.counting()));
        System.out.println("Total Male: " + genderCount.getOrDefault("Male", 0L) + " Female:" + genderCount.getOrDefault("Female", 0L));


        Set<String> departments = employeeList.stream().map(employee -> employee.getDepartment()).collect(Collectors.toSet());
        System.out.println(departments);


        Map<String, Double> averageAge = employeeList.stream().collect(Collectors.groupingBy(employee -> employee.getGender(), Collectors.averagingInt(employee -> employee.getAge())));
        System.out.println("Total Average of male:" + averageAge.getOrDefault("Male", 0.0) + " Average of Female:" + averageAge.getOrDefault("Female", 0.0));

        Optional<Employee> max = employeeList.stream().max(Comparator.comparingDouble(employee -> employee.getSalary()));
        max.ifPresent(employee -> System.out.println("Highest Salaried Employee name: " + employee.getName()));

        List<Employee> employeeJoinedAfter2015 = employeeList.stream()
                .filter(employee -> employee.getYearOfJoining() > 2015).collect(Collectors.toList());
        System.out.println("Employees Joined After 2015: ");
        employeeJoinedAfter2015.forEach(employee -> System.out.println(employee.getName()));

        Map<String, Long> departmentWiseEmployeeCount = employeeList.stream()
                .collect(Collectors.groupingBy(employee -> employee.getDepartment(), Collectors.counting()));
        System.out.println(departmentWiseEmployeeCount);

        Map<String, Double> departmentWiseAverageSalary = employeeList.stream()
                .collect(Collectors.groupingBy(employee -> employee.getDepartment(), Collectors.averagingDouble(employee -> employee.getSalary())));
        System.out.println(departmentWiseAverageSalary);

        Optional<Employee> youngestEmployeeInDepartment = employeeList.stream()
                .filter(employee -> employee.getDepartment().equals("Engineering") && employee.getGender().equals("Male"))
                .min(Comparator.comparingInt(employee -> employee.getAge()));
        youngestEmployeeInDepartment.ifPresent(employee -> System.out.println(employee.getName()));

        Map<String, Long> maleFemaleInDepartment = employeeList.stream()
                .filter(employee -> employee.getDepartment().equals("Sales") || employee.getDepartment().equals("Marketing"))
                .collect(Collectors.groupingBy(employee -> employee.getGender(), Collectors.counting()));

        System.out.println("Male count in sales: " + maleFemaleInDepartment.getOrDefault("Male", 0L));

        Map<String, Double> maleFemaleAverageSalary = employeeList.stream()
                .collect(Collectors.groupingBy(Employee::getGender, Collectors.averagingDouble(Employee::getSalary)));

        System.out.println(maleFemaleAverageSalary);

        Map<String, List<String>> departmentEmployeeName = employeeList.stream()
                .collect(Collectors.groupingBy(Employee::getDepartment,
                        Collectors.mapping(Employee::getName, Collectors.toList())));

        System.out.println(departmentEmployeeName);

        DoubleSummaryStatistics salaryStats = employeeList.stream()
                .collect(Collectors.summarizingDouble(Employee::getSalary));

        System.out.println("Average Salary: " + salaryStats.getAverage());
        System.out.println("Total Salary: " + salaryStats.getSum());

        Map<Boolean, List<Employee>> partitionedByAge = employeeList.stream()
                .collect(Collectors.partitioningBy(employee -> employee.getAge() <= 25));

        System.out.println("Employees ≤ 25 years: " + partitionedByAge.get(true));
        System.out.println("Employees > 25 years: " + partitionedByAge.get(false));

        Optional<Employee> oldestEmployee = employeeList.stream().max(Comparator.comparingInt(Employee::getAge));

        oldestEmployee.ifPresent(emp ->
                System.out.println("Oldest Employee: " + emp.getName() +
                        ", Age: " + emp.getAge() +
                        ", Department: " + emp.getDepartment()));

        List<Employee> employeesNameSorted = employeeList.stream()
                .sorted(Comparator.comparing(Employee::getName))
                .collect(Collectors.toList());

        System.out.println(employeesNameSorted);

        Optional<Employee> secondHighestSalary = employeeList.stream()
                .sorted(Comparator.comparingDouble(Employee::getSalary))
                .distinct()
                .skip(1)
                .findFirst();

        secondHighestSalary.ifPresentOrElse(
                employee -> System.out.println("Second Highest salary: " + employee.getSalary()),
                () -> System.out.println("Not found employee with second highest salary"));


    }
}
